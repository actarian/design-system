## Example

A basic example of `Drawer`.

export const Basic = ({ data, ...args }) => {
  const [showDrawer, setShowDrawer] = useState(false);
  const onOpenDrawer = () => setShowDrawer(true);
  const onCloseDrawer = () => {
    setShowDrawer(false);
  }
  return (<>
    <div style={{ margin: '0 1em' }}>
      <Button type="default" marginBottom="1rem" onClick={onOpenDrawer}>Open Drawer</Button>
      <Drawer visible={showDrawer} onClose={() => onCloseDrawer()} placement="right">
        <Drawer.Title>Drawer</Drawer.Title>
        <Drawer.Subtitle>This is a drawer</Drawer.Subtitle>
        <Drawer.Content>
          <p>Some content contained within the drawer.</p>
        </Drawer.Content>
      </Drawer>
    </div>
  </>);
}
