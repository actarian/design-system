## Example

A basic example of `Modal`.

export const Basic = ({ data, ...args }) => {
  const [showModal, setShowModal] = useState(false);
  const onOpenModal = () => setShowModal(true);
  const onCloseModal = () => {
    setShowModal(false);
  }
  return (<>
    <div style={{ margin: '0 1em' }}>
      <Button type="default" marginBottom="1rem" onClick={onOpenModal}>Open Modal</Button>
      <Modal width="30rem" visible={showModal} onClose={onCloseModal}>
        <Modal.Title>
          <Text type="8" fontWeight="700">Modal Title</Text>
        </Modal.Title>
        <Modal.Subtitle>Modal Subtitle</Modal.Subtitle>
        <Modal.Content>
          <p>Modal content lorem ispum.</p>
        </Modal.Content>
        <Modal.Button passive onClick={onCloseModal}>Cancel</Modal.Button>
        <Modal.Button>Confirm</Modal.Button>
      </Modal>
    </div>
  </>);
}
